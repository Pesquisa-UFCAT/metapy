#### Example 01

```python 
from META_TOOLBOX import HILL_CLIMBING_001

# Input
SETUP = {
        'N_REP': 10,
        'N_POP': 5,
        'N_ITER': 1000,
        'X_L': [-2, -2, -2],
        'X_U': [2, 2, 2],
        'D': 3,
        'SIGMA': 0.10,
        'NULL_DIC': None
        }

# OF statement
def OF_FUNCTION(X, NULL_DIC):
    X_0 = X[0]
    X_1 = X[1]
    X_2 = X[2]
    OF = X_0 ** 2 + X_1 ** 2 + X_2 ** 2
    return OF

# Call function
RESULTS_REP, BEST_REP, AVERAGE_REP, WORST_REP, STATUS_PROCEDURE = HILL_CLIMBING_001(OF_FUNCTION, SETUP)
```
